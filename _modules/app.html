@app.route('/api/items/<int:item_id>', methods=['GET'])
def get_item(item_id):
    """
    특정 아이템 상세 정보 조회
    ---
    parameters:
      - name: item_id
        in: path
        type: integer
        required: true
        description: 조회할 아이템의 ID
    responses:
      200:
        description: 아이템 정보 반환
      404:
        description: 아이템을 찾을 수 없음
    """
    item = next((item for item in items_db if item["id"] == item_id), None)
    if item:
        return jsonify(item), 200
    return jsonify({"error": "Item not found"}), 404

@app.route('/api/items', methods=['POST'])
def add_item():
    """
    새로운 아이템 추가
    ---
    parameters:
      - name: body
        in: body
        required: true
        schema:
          id: Item
          required:
            - name
            - price
          properties:
            name:
              type: string
              description: 아이템 이름
            price:
              type: integer
              description: 가격
    responses:
      201:
        description: 아이템 생성 완료
    """
    new_data = request.json
    new_item = {
        "id": len(items_db) + 1,
        "name": new_data.get("name"),
        "price": new_data.get("price")
    }
    items_db.append(new_item)
    return jsonify(new_item), 201

@app.route('/api/items/<int:item_id>', methods=['PUT'])
def update_item(item_id):
    """
    기존 아이템 정보 수정
    ---
    parameters:
      - name: item_id
        in: path
        type: integer
        required: true
      - name: body
        in: body
        required: true
        schema:
          properties:
            name:
              type: string
            price:
              type: integer
    responses:
      200:
        description: 수정 성공
    """
    item = next((item for item in items_db if item["id"] == item_id), None)
    if not item:
        return jsonify({"error": "Item not found"}), 404
    
    update_data = request.json
    item.update(update_data)
    return jsonify(item), 200

@app.route('/api/items/<int:item_id>', methods=['DELETE'])
def delete_item(item_id):
    """
    특정 아이템 삭제
    ---
    parameters:
      - name: item_id
        in: path
        type: integer
        required: true
    responses:
      200:
        description: 삭제 성공
    """
    global items_db
    items_db = [item for item in items_db if item["id"] != item_id]
    return jsonify({"message": "Item deleted"}), 200
